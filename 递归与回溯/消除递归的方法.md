- 尾递归转循环：递归调用出现在函数末尾，可用循环结构代替递归
- 利用栈模拟递归过程：具体问题具体分析，结合一定技巧

## 尾递归转循环
- 尾递归：递归调用出现在函数末尾，可用循环结构代替递归
- 示例：
  - 递归代码：
    ```cpp {.line-numbers}
    int f(int n, int sum){
        if(n == 0) return sum;
        sum += n;
    return f(n - 1, sum);
    }
    ```
  - 循环代码：
    ```cpp {.line-numbers}
    int f(int n, int sum){
        while(n > 0){ 
            sum += n; 
            n--;
        }
        return sum;
    }
    ```

## 利用自定义栈模拟递归过程（以汉诺塔问题为例）
- 汉诺塔问题说明：借助B柱，将n个盘子从A柱移至C柱
- 递归算法：
  ```cpp {.line-numbers}
    void Hanoi(int n, char A, char B, char C){
        //将 n 个圆盘由A柱经B柱移至C柱
        if(n == 1) move(A, C); // 将 A 柱唯一圆盘移至 C 柱
        else{
            Hanoi(n - 1, A, C, B);
            move(A, C);
            Hanoi(n - 1, B, A, C);
        }
    }

    void move(char x, char y){
        printf("Move disk from %c to %c\n", x, y);
    }
  ```
- 利用自定义栈模拟递归过程：
  - 核心思路：按处理子任务的逆序进栈
    - 先处理的子任务后入栈
    - 后处理的子任务先入栈
  - 示例代码：
  ```cpp {.line-numbers}
    struct Quad { int n; char A, B, C; }; // 四元组
    class HnoiStack{
        private:
            Quad H[MaxSize];
            int top = -1;
        public:
            bool empty() {return top == -1;}
            bool full() {return top == MaxSize - 1;}
            void push(int n, char A, char B, char C){
                assert(!full());    top++;
                H[top].n = n; H[top].A = A; H[top].B = B; H[top].C = C;
            }
            void pop(int &n, char &A, char &B, char &C){
                assert(!empty());
                n = H[top].n;   A = H[top].A;   B = B[top].B;   C = H[top].C;
                top--;
            }
    }

    // Hanoi塔问题的非递归算法
    void Hanoi_NonRec(int n, char A, char B, char C){
        HnoiStack s;
        s.push(n, A, B, C);
        while(!s.empty()){
            int n;  char A, B, C;
            s.pop(n, A, B, C);
            if(n == 1) move(A, C);
            else{
                s.push(n - 1, B, A, C);
                s.push(1, A, B, C);
                s.push(n - 1, A, C, B);
            }
        }
    }
  ```
